<script>
  const player = document.getElementById("player");
  const hurdle = document.getElementById("hurdle");
  const coin = document.getElementById("coin");
  const scoreEl = document.getElementById("score");
  const coinsEl = document.getElementById("coins");
  const gameOverScreen = document.getElementById("gameOver");

  let score = 0;
  let coins = 0;
  let isJumping = false;
  let gameOver = false;

  function jump() {
    if (isJumping || gameOver) return;
    isJumping = true;
    player.classList.add("jump");
    setTimeout(() => {
      player.classList.remove("jump");
      isJumping = false;
    }, 600);
  }

  document.addEventListener("keydown", jump);
  document.addEventListener("touchstart", jump);

  setInterval(() => {
    if (gameOver) return;

    score++;
    scoreEl.innerText = score;

    const playerRect = player.getBoundingClientRect();
    const hurdleRect = hurdle.getBoundingClientRect();
    const coinRect = coin.getBoundingClientRect();

    // Collision with hurdle
    if (
      hurdleRect.left < playerRect.right &&
      hurdleRect.right > playerRect.left &&
      hurdleRect.bottom > playerRect.top
    ) {
      gameOver = true;
      hurdle.style.animation = "none";
      coin.style.animation = "none";
      gameOverScreen.style.display = "flex";
    }

    // Coin collection
    if (
      coinRect.left < playerRect.right &&
      coinRect.right > playerRect.left &&
      coinRect.bottom > playerRect.top &&
      coinRect.top < playerRect.bottom
    ) {
      coins++;
      coinsEl.innerText = coins;
      coin.style.display = "none";
      setTimeout(() => coin.style.display = "block", 1000);
    }

  }, 100);
</script>
